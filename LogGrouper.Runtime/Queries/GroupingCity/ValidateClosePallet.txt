SELECT DISTINCT O.ORDERKEY as OrderId, ADP.DROPID AS DropId  
FROM {Scheme}.AGRUPADO_DETALLE_PALLET ADP
JOIN {Scheme}.PICKDETAIL PD ON ADP.ID_PREPRACION = PD.DROPID
JOIN {Scheme}.ORDERS O ON PD.ORDERKEY = O.ORDERKEY
WHERE ADP.DROPID = '{Drop_Id}'
AND O.ORDERKEY NOT IN (SELECT ADP.ORDERKEY FROM {Scheme}.AGRUPADO_DETALLE_PALLET ADP WHERE DROPID = '{Drop_Id}')
