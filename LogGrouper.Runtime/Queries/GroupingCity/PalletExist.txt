SELECT AD.DROPID AS DROPID, 
       AD.USUARIO AS USUARIO,  
       ISNULL(O.REFERENCEDOCUMENT, AD.ORDERKEY) AS NUMEROPEDIDO,
       D.ISCLOSED,
       O.STATUS,
       O.ORDERGROUP,
       PD.DROPID AS LPN
FROM {Scheme}.ORDERS O
LEFT JOIN {Scheme}.PICKDETAIL PD ON PD.ORDERKEY = O.ORDERKEY
LEFT JOIN {Scheme}.AGRUPADO_DETALLE_PALLET AD ON O.ORDERKEY = AD.ORDERKEY
LEFT JOIN {Scheme}.DROPID D ON AD.DROPID = D.DROPID
WHERE O.ORDERKEY in({Order_Id})
